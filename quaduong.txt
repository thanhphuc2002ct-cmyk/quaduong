// code qua đường có các chức năng của led RGB và còi, còi sẽ kêu +1 mỗi khi tiến thêm 1 vạch.
// khi gặp line kết thúc nháy đèn 10 lần đồng thời kêu tít 10 lần rồi chạy tiếp.
// lưu ý xe chỉ chớp đèn khi thấy được line bắt đầu là full line rồi tới line chính giữa.
#include <Arduino.h>
#include "config.h"
#include "motor.h"
#include "master.h"
#include <Adafruit_NeoPixel.h> 

// --- KHỞI TẠO NEOPIXEL ---
Adafruit_NeoPixel pixels(NUMPIXELS, LED_PIN, NEO_GRB + NEO_KHZ800);

// ====================================================================
// --- MODULE CHỈ BÁO (ĐÈN & CÒI) ĐÃ ĐƯỢC GỘP VÀO ---
// ====================================================================
enum IndState { IDLE, BEEPING_STRIPE, IND_BLINKING_END }; // Đổi tên để không trùng với trạng thái của xe
IndState indState = IDLE;

unsigned long indPreviousMillis = 0;
int indBlinkCount = 0;
bool indLedState = false;

// Biến cho còi kêu nhiều nhịp
int targetBeeps = 0;
int currentBeepCount = 0;
bool isBeepOn = false;

void indicatorClear() {
    pixels.clear();
    pixels.show();
    digitalWrite(BUZZER_PIN, LOW);
    indState = IDLE;
}

void indicatorInit() {
    pinMode(BUZZER_PIN, OUTPUT);
    digitalWrite(BUZZER_PIN, LOW);
    
    pixels.begin();
    pixels.clear();
    pixels.show();
}

void indicatorShowStripe(int count) {
    pixels.clear();
    if (count == 1) pixels.setPixelColor(0, pixels.Color(255, 0, 0));
    else if (count == 2) pixels.setPixelColor(0, pixels.Color(0, 255, 0));
    else if (count == 3) pixels.setPixelColor(0, pixels.Color(255, 255, 0));
    else if (count == 4) pixels.setPixelColor(0, pixels.Color(0, 0, 255));
    else if (count == 5) pixels.setPixelColor(0, pixels.Color(128, 0, 128));
    else if (count >= 6) pixels.setPixelColor(0, pixels.Color(0, 255, 255));
    pixels.show();

    // Thiết lập còi kêu số lần tương ứng với số vạch
    targetBeeps = count;
    currentBeepCount = 0;
    isBeepOn = true;
    digitalWrite(BUZZER_PIN, HIGH);
    
    indState = BEEPING_STRIPE;
    indPreviousMillis = millis();
}

void indicatorStartEndBlink() {
    indState = IND_BLINKING_END;
    indBlinkCount = 0;
    indLedState = true;
    indPreviousMillis = millis();
    
    // Sáng đèn + kêu còi ngay nhịp đầu tiên
    pixels.setPixelColor(0, pixels.Color(255, 255, 255));
    pixels.show();
    digitalWrite(BUZZER_PIN, HIGH); 
}

bool isIndicatorBusy() {
    return (indState == IND_BLINKING_END);
}

void indicatorUpdate() {
    unsigned long currentMillis = millis();

    // 1. Kêu còi nhiều nhịp
    if (indState == BEEPING_STRIPE) {
        if (currentMillis - indPreviousMillis >= 100) {
            indPreviousMillis = currentMillis;
            
            if (isBeepOn) {
                digitalWrite(BUZZER_PIN, LOW);
                isBeepOn = false;
                currentBeepCount++;
                
                // Nếu đã kêu đủ số lần thì tắt trạng thái bíp
                if (currentBeepCount >= targetBeeps) {
                    indState = IDLE;
                }
            } else {
                digitalWrite(BUZZER_PIN, HIGH);
                isBeepOn = true;
            }
        }
    }
    // 2. Chớp đèn trắng và kêu còi 10 lần (20 chu kỳ sáng/tắt)
    else if (indState == IND_BLINKING_END) {
        if (currentMillis - indPreviousMillis >= 500) {
            indPreviousMillis = currentMillis;
            indLedState = !indLedState; 

            if (indLedState) {
                pixels.setPixelColor(0, pixels.Color(255, 255, 255));
                digitalWrite(BUZZER_PIN, HIGH);
            } else {
                pixels.clear();
                digitalWrite(BUZZER_PIN, LOW);
            }
            pixels.show();
            
            indBlinkCount++;
            if (indBlinkCount >= 20) { // 10 lần chớp = 20 nhịp
                indicatorClear(); 
            }
        }
    }
}


// ====================================================================
// --- PHẦN LOGIC CHÍNH CỦA XE ---
// ====================================================================

Master comm; 

uint16_t stripe_count = 0;
uint8_t last_val = 255;
bool saw_mark_27 = 0; 
bool is_stopped = 0;
bool enable_led = 0; 

// --- CÁC BIẾN CHO MILLIS() ---
unsigned long previousMillis = 0;
unsigned long lastI2CPoll = 0; 

// --- ĐỊNH NGHĨA TRẠNG THÁI CHẠY XE ---
enum RobotState {
    RUNNING,          // Đang dò line
    PAUSING_STRIPE,   // Đang dừng 3s ở vạch sọc
    PAUSING_END,      // Đang dừng 10s ở vạch kết thúc (không có đèn)
    BLINKING_END,     // Đang chờ báo hiệu kết thúc hoàn tất
    CROSSING_LINE     // Đang bơm ga 400ms để vọt qua vạch ngang
};
RobotState currentState = RUNNING;

void setup() {
    Serial.begin(115200);
    comm.beginI2C(SDA_PIN, SCL_PIN);
    motorInit();
    
    // Khởi tạo đèn + còi (Hàm nội bộ)
    indicatorInit(); 
}

void loop() {
    // LUÔN DUY TRÌ HÀM NÀY ĐỂ ĐÈN VÀ CÒI CHẠY NGẦM BẰNG MILLIS
    indicatorUpdate();

    if (is_stopped) return;
    unsigned long currentMillis = millis();

    switch (currentState) {
        
        case PAUSING_STRIPE:
            if (currentMillis - previousMillis >= 3000) {
                currentState = CROSSING_LINE; 
                setMotors(45, 40);
                previousMillis = currentMillis;
            }
            return; 

        case PAUSING_END:
            if (currentMillis - previousMillis >= 10000) {
                currentState = CROSSING_LINE;
                setMotors(45, 40);
                previousMillis = currentMillis;
            }
            return;

        case BLINKING_END:
            // Chỉ cần đứng đợi đến khi đèn còi báo bận xong (chớp đủ 10 lần) là vọt tiếp
            if (!isIndicatorBusy()) {
                enable_led = false;
                currentState = CROSSING_LINE;
                setMotors(45, 40);
                previousMillis = currentMillis;
            }
            return;

        case CROSSING_LINE:
            if (currentMillis - previousMillis >= 400) {
                currentState = RUNNING; 
            }
            return; 

        case RUNNING:
            if (currentMillis - lastI2CPoll < 10) return;
            lastI2CPoll = currentMillis;

            uint8_t raw_val = 255; 
            comm.I2CrequestFrom(I2C_ADDR, 1, &raw_val);
            if (raw_val == 255) return; 

            uint8_t val = raw_val & 0x1F; 

            if (val == START_MARK_VAL && !enable_led) {
                enable_led = true;
                Serial.println("Da gap vach dao nguoc -> CHO PHEP SANG DEN");
            }

            if (val == 27) saw_mark_27 = 1; 
            else if (val == 31) saw_mark_27 = 0; 

            if (val == 0 && last_val != 0) { 
                setMotors(0, 0); 
                
                if (saw_mark_27 == 1) {
                    Serial.println("NGUNG: Da thay vach nho ra 27 roi moi toi 0");
                    stripe_count = 0; // Đặt lại biến đếm khi gặp vạch kết thúc

                    if (enable_led) {
                        currentState = BLINKING_END;
                        indicatorStartEndBlink(); 
                    } else {
                        currentState = PAUSING_END;
                        previousMillis = currentMillis; 
                    }
                    saw_mark_27 = 0; 
                } 
                else {
                    // CHỈ ĐẾM VÀ KÊU CÒI KHI ĐÃ ĐƯỢC PHÉP SÁNG ĐÈN
                    if (enable_led) {
                        stripe_count++;
                        Serial.printf("Dem vach soc: %d - Keu %d tieng (Ngung 3s)\n", stripe_count, stripe_count);
                        indicatorShowStripe(stripe_count); 
                    } else {
                        Serial.println("Gap vach soc nhung chua dem (Ngung 3s)");
                    }
                    
                    currentState = PAUSING_STRIPE;
                    previousMillis = currentMillis; 
                }
            } 
            else if (val != 0) {
                setMotors(45, 40);
            }

            last_val = val;
            break; 
    }
}
